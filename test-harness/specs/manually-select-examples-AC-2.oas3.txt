====spec====
openapi: 3.0.2
paths:
  /todos:
    get:
      responses:
        200:
          description: Get Todo Items
          "content":
            application/json:
              schema:
                type: number

mock -p 4010

curl -i http://localhost:4010/todos?__example=mantis-shrimp

HTTP/1.1 404 Not Found
content-type: application/problem+json

{"type":"https://stoplight.io/prism/errors#NOT_FOUND","title":"The server cannot find the requested content","status":404,"detail":"Response for contentType: application/json and exampleKey: mantis-shrimp does not exist."}
