====test====
If I have an OAS3 document referencing the same model twice
it should return 200 on a regular request and not crash
====spec====
openapi: 3.0.0
paths:
  /:
    get:
      responses:
        '200':
          content:
            application/json:
              schema:
                type: object
                properties:
                  first:
                    $ref: '#/components/schemas/TestSchema'
                  second:
                    $ref: '#/components/schemas/TestSchema'
components:
  schemas:
    TestSchema:
      type: object
      properties:
        output:
          type: string
          nullable: true
====server====
mock -p 4010 ${document}
====command====
curl -i http://localhost:4010/
====expect====
HTTP/1.1 200 OK

{"first":{"output":"string"},"second":{"output":"string"}}
