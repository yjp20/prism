====test====
Making a request to an operation with removed global security results in granted access.
====spec====
swagger: '2.0'
paths:
  /todos:
    get:
      produces:
        - application/json
      security: []
      responses:
        200:
          description: OK
          examples:
            application/json: test
securityDefinitions:
  BasicAuth:
    type: basic
  ApiKeyAuth:
    type: apiKey
    in: header
    name: X-API-Key
security:
  - BasicAuth: []
  - ApiKeyAuth: []
====server====
mock -p 4010
====command====
curl -i http://localhost:4010/todos
====expect====
HTTP/1.1 200 OK

"test"
